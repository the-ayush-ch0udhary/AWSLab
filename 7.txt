Write a program to develop a Concurrent chat server using TCP socket. Use at least two client programs to implement it.

server

#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <stdlib.h>

#define SERVER_IP "127.0.0.1"
#define SERVER_PORT 4560

int main(){
    int sd, nsd, pid, clen;
    struct sockaddr_in server, client;
    char msg[512];

    sd = socket(AF_INET, SOCK_STREAM, 0);

    bzero((char *)&server, sizeof(server));
    server.sin_family = AF_INET;
    server.sin_addr.s_addr = inet_addr(SERVER_IP);
    server.sin_port = htons(SERVER_PORT);

    bind(sd, (struct sockaddr *)&server, sizeof(server));
    listen(sd, 5);

    while(1){
        clen = sizeof(client);
        nsd = accept(sd, (struct sockaddr *)&client, &clen);

        pid = fork();

        if(pid == 0){
            // CHILD handles one client
            while(1){
                memset(msg, 0x0, 512);
                read(nsd, msg, 512,0);

                printf("Client: %s\n", msg);

                if(strcmp(msg, "stop") == 0)
                    break;

                printf("Server: ");
                scanf("%s", msg);

                write(nsd, msg, strlen(msg)+1);

                if(strcmp(msg, "stop") == 0)
                    break;
            }

            close(nsd);
            exit(0);
        }
        else{
            // PARENT continues for next clients
            close(nsd);
        }
    }

    close(sd);
    return 0;
}
