Write a program to develop a Concurrent Day-Time server using TCP socket. Use at
least two client programs to implement it.

server

#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <stdlib.h>
#include <time.h>

#define SERVER_IP "127.0.0.1"
#define SERVER_PORT 5005

int main(){
    int sd, nsd, pid, clen;
    struct sockaddr_in server, client;
    char msg[512];
    time_t t;
    char *ptr;

    sd = socket(AF_INET, SOCK_STREAM, 0);

    bzero((char *)&server, sizeof(server));
    server.sin_family = AF_INET;
    server.sin_addr.s_addr = inet_addr(SERVER_IP);
    server.sin_port = htons(SERVER_PORT);

    bind(sd, (struct sockaddr *)&server, sizeof(server));
    listen(sd, 5);

    while(1){
        clen = sizeof(client);
        nsd = accept(sd, (struct sockaddr *)&client, &clen);

        pid = fork();

        if(pid == 0){
            // CHILD handles client
            while(1){
                memset(msg, 0, 512);
                read(nsd, msg, 512);

                if(strcmp(msg, "time") == 0){
                    t = time(NULL);
                    ptr = ctime(&t);

                    write(nsd, ptr, strlen(ptr)+1);
                }
                else if(strcmp(msg, "stop") == 0){
                    break;
                }
                else{
                    strcpy(msg, "Invalid request");
                    write(nsd, msg, strlen(msg)+1);
                }
            }

            close(nsd);
            exit(0);
        }
        else{
            // PARENT handles next client
            close(nsd);
        }
    }

    close(sd);
    return 0;
}

client

#include <stdio.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <sys/socket.h>

#define SERVER_IP "127.0.0.1"
#define SERVER_PORT 5005

int main(){
    int sd;
    struct sockaddr_in server;
    char msg[512];

    sd = socket(AF_INET, SOCK_STREAM, 0);

    bzero((char *)&server, sizeof(server));
    server.sin_family = AF_INET;
    server.sin_addr.s_addr = inet_addr(SERVER_IP);
    server.sin_port = htons(SERVER_PORT);

    connect(sd, (struct sockaddr *)&server, sizeof(server));

    while(1){
        printf("Enter message (time/stop): ");
        scanf("%s", msg);

        write(sd, msg, strlen(msg)+1);

        if(strcmp(msg, "stop") == 0)
            break;

        read(sd, msg, 512);
        printf("Server time: %s\n", msg);
    }

    close(sd);
    return 0;
}
